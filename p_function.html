def eta1(w,q):
    z=0
    for n in range(1,8):
        z-=24*n*q**(2*n)/(1-q**(2*n))
    z=z+1
    z=z*pi*pi/3/w
    return z

def theta1(v,t):
    q=mp.exp(1j*pi*t)
    ans=0j
    for n in range(8):
        ans+=2*(-1)**n*q**(n**2+n+1/4)*mp.sin((2*n+1)*pi*v)
    return ans

def dtheta1(v,t):
    q=mp.exp(1j*pi*t)
    ans=0j
    for n in range(8):
        ans+=2*(2*n+1)*pi*(-1)**n*q**(n**2+n+1/4)*mp.cos((2*n+1)*pi*v)
    return ans

def ddtheta1(v,t):
    q=mp.exp(1j*pi*t)
    ans=0j
    for n in range(8):
        ans+=2*(2*n+1)**2*pi**2*(-1)**(n+1)*q**(n**2+n+1/4)*mp.sin((2*n+1)*pi*v)
    return ans

def theta2(v,t):
    q=mp.exp(1j*pi*t)
    ans=0j
    for n in range(8):
        ans+=2*q**(n**2+n+1/4)*mp.cos((2*n+1)*pi*v)
    return ans

def theta3(v,t):
    q=mp.exp(1j*pi*t)
    ans=1
    for n in range(1,8):
        ans+=2*q**(n**2)*mp.cos(2*n*pi*v)
    return ans

def theta0(v,t):
    q=mp.exp(1j*pi*t)
    ans=1
    for n in range(1,8):
        ans+=2*(-1)**n*q**(n**2)*mp.cos(2*n*pi*v)
    return ans

def pe(u,w1,w2):
    tau=w2/w1
    q=mp.exp(1j*pi*tau)
    et=(eta1(w1,q))
    ep=1e-200
    uu=u/w1
    a=theta1(uu,tau)
    b=dtheta1(uu,tau)
    c=ddtheta1(uu,tau)
    return -et/w1+1/(w1**2)*(b**2/(a**2+ep)-c/(a+ep))

def sn(u,t):
    K=pi/2*theta3(0,t)**2
    return theta3(0,t)/theta2(0,t)*theta1(u/K*0.5,t)/theta0(u/K*0.5,t)

def cn(u,t):
    K=pi/2*theta3(0,t)**2
    return theta0(0,t)/theta2(0,t)*theta2(u/K*0.5,t)/theta0(u/K*0.5,t)

def dn(u,t):
    K=pi/2*theta3(0,t)**2
    return theta0(0,t)/theta3(0,t)*theta3(u/K*0.5,t)/theta0(u/K*0.5,t)
